1) Blueprint format (what to store as JSON/YAML per module)

Use one module = one blueprint. Store it as JSON (or YAML) and keep it database-agnostic, but with enough detail to generate SQL.

Blueprint JSON shape (core)

2) The “Blueprint Registry” SQL database (stores + versions these blueprints)

This is the database to run personal tool. It stores modules, versions, configs, and optionally compiled outputs (SQL).

Core tables (DDL
-- 1) Module identity
CREATE TABLE blueprint_modules (
  id              uuid PRIMARY KEY,
  module_key      text NOT NULL UNIQUE,        -- e.g. "marketplace.core"
  name            text NOT NULL,
  description     text,
  created_at      timestamptz NOT NULL DEFAULT now(),
  updated_at      timestamptz NOT NULL DEFAULT now()
);

-- 2) Module versions (immutable)
CREATE TABLE blueprint_versions (
  id              uuid PRIMARY KEY,
  module_id       uuid NOT NULL REFERENCES blueprint_modules(id),
  version         text NOT NULL,               -- "1.0.0"
  blueprint_json  jsonb NOT NULL,              -- full blueprint document
  checksum        text NOT NULL,               -- hash for integrity
  is_deprecated   boolean NOT NULL DEFAULT false,
  created_at      timestamptz NOT NULL DEFAULT now(),
  UNIQUE (module_id, version)
);

-- 3) Dependency graph per version (denormalized from JSON for quick resolution)
CREATE TABLE blueprint_dependencies (
  id                    uuid PRIMARY KEY,
  blueprint_version_id   uuid NOT NULL REFERENCES blueprint_versions(id) ON DELETE CASCADE,
  depends_on_module_key  text NOT NULL,
  version_range          text NOT NULL,
  optional              boolean NOT NULL DEFAULT false
);

-- 4) Named “packs” (a project’s chosen set of modules + config)
CREATE TABLE blueprint_packs (
  id              uuid PRIMARY KEY,
  pack_key        text NOT NULL UNIQUE,        -- e.g. "my-marketplace-ai"
  name            text NOT NULL,
  description     text,
  created_at      timestamptz NOT NULL DEFAULT now(),
  updated_at      timestamptz NOT NULL DEFAULT now()
);

CREATE TABLE pack_modules (
  id              uuid PRIMARY KEY,
  pack_id         uuid NOT NULL REFERENCES blueprint_packs(id) ON DELETE CASCADE,
  module_key      text NOT NULL,
  version         text NOT NULL,               -- pinned version
  sort_order      int NOT NULL DEFAULT 0,
  UNIQUE (pack_id, module_key)
);

-- 5) Pack config (the resolved config inputs you pass to generator)
CREATE TABLE pack_config (
  id              uuid PRIMARY KEY,
  pack_id         uuid NOT NULL REFERENCES blueprint_packs(id) ON DELETE CASCADE,
  config_json     jsonb NOT NULL,              -- e.g. { "multi_tenant": true, ... }
  created_at      timestamptz NOT NULL DEFAULT now()
);

-- 6) Build outputs (compiled artifacts)
CREATE TABLE pack_builds (
  id              uuid PRIMARY KEY,
  pack_id         uuid NOT NULL REFERENCES blueprint_packs(id) ON DELETE CASCADE,
  build_number    int NOT NULL,
  resolved_manifest jsonb NOT NULL,            -- dependency-resolved, conditions applied
  sql_migration   text NOT NULL,               -- generated SQL
  generated_at    timestamptz NOT NULL DEFAULT now(),
  UNIQUE (pack_id, build_number)
);


This gives you:
	•	Versioned blueprints (immutable)
	•	Dependency resolution tables (fast)
	•	“Pack” concept for real projects
	•	Stored generated SQL migrations and manifests

3) “Runtime schema” SQL for AI-native attach points (what generated DB includes)

When blueprints include AI capabilities + attach points, I want the generator to optionally include these universal tables.

AI Core (module: ai.core)

CREATE TABLE ai_runs (
  id              uuid PRIMARY KEY,
  entity_type     text NOT NULL,   -- "listings"
  entity_id       uuid NOT NULL,
  capability_key  text NOT NULL,   -- "semantic_search" / "content_enhancement"
  model           text NOT NULL,
  input_payload   jsonb NOT NULL,
  output_payload  jsonb,
  status          text NOT NULL,   -- "queued"|"success"|"error"
  cost_usd        numeric(12,6),
  latency_ms      int,
  created_at      timestamptz NOT NULL DEFAULT now()
);

CREATE TABLE ai_feedback (
  id          uuid PRIMARY KEY,
  ai_run_id   uuid NOT NULL REFERENCES ai_runs(id) ON DELETE CASCADE,
  rating      int,          -- e.g. 1..5
  correction  text,
  created_at  timestamptz NOT NULL DEFAULT now()
);

-- For embedding targets (semantic search)
CREATE TABLE embeddings (
  id            uuid PRIMARY KEY,
  entity_type   text NOT NULL,     -- "listings"
  entity_id     uuid NOT NULL,
  model         text NOT NULL,
  vector        bytea NOT NULL,    -- store as bytea, or use pgvector if you choose
  metadata      jsonb,
  created_at    timestamptz NOT NULL DEFAULT now()
);

CREATE INDEX embeddings_entity_idx ON embeddings(entity_type, entity_id);

Attach points (files/comments) as universal modules

Attachments module

4) Events layer (module: core.events) for automation + MCP-friendly integration

This is the “backbone” for internal automation tools.

CREATE TABLE events (
  id            uuid PRIMARY KEY,
  event_name    text NOT NULL,           -- "listing.created"
  entity_type   text,
  entity_id     uuid,
  payload       jsonb NOT NULL,
  occurred_at   timestamptz NOT NULL DEFAULT now()
);

CREATE INDEX events_name_time_idx ON events(event_name, occurred_at);
CREATE INDEX events_entity_idx ON events(entity_type, entity_id);

CREATE TABLE event_subscriptions (
  id              uuid PRIMARY KEY,
  subscriber_key  text NOT NULL,         -- "automation.workflow_engine" or "ai.agent_router"
  event_name      text NOT NULL,
  filter_json     jsonb,                 -- optional filtering rules
  is_enabled      boolean NOT NULL DEFAULT true,
  created_at      timestamptz NOT NULL DEFAULT now()
);

CREATE TABLE event_deliveries (
  id                uuid PRIMARY KEY,
  event_id           uuid NOT NULL REFERENCES events(id) ON DELETE CASCADE,
  subscription_id    uuid NOT NULL REFERENCES event_subscriptions(id) ON DELETE CASCADE,
  status             text NOT NULL,       -- "pending"|"delivered"|"failed"
  attempts           int NOT NULL DEFAULT 0,
  last_error         text,
  delivered_at       timestamptz
);

This supports:
	•	workflow triggers
	•	AI agent triggers
	•	audit trails

5) How the MCP Server uses this (practical contract)

Your MCP server typically needs three things:

A) Read blueprint modules
	•	GET /blueprints/modules
	•	GET /blueprints/module/{module_key}/versions
	•	GET /blueprints/version/{id} (returns JSON)

B) Build a pack (resolve dependencies + apply config)
	•	POST /packs (create pack)
	•	POST /packs/{pack_key}/build (produces resolved manifest + SQL migration)

C) Publish runtime metadata for no-code UI

From the resolved manifest, emit:
	•	entities list (table names, fields, types)
	•	relationships graph
	•	events catalog (event names + payload schema)
	•	attach points
	•	ai capabilities enabled

That metadata becomes your “No-code backend model layer.”

MCP server generator pipeline design

This is the end-to-end engine that takes a Pack and produces:
	•	SQL schema/migrations
	•	resolved manifest metadata (for your no-code backend)
	•	runtime service wiring (rules/workflows/agents/tools)

3A) Core pipeline stages

Stage 0 — Inputs
	•	pack_key
	•	module list (module_key + version)
	•	pack config (merged config)
	•	target SQL dialect (Postgres recommended)

Stage 1 — Load modules
	•	Fetch blueprint JSON for each module/version.
	•	Validate JSON schema for your blueprint format.

Stage 2 — Resolve dependencies
	•	Build dependency graph from dependencies[].
	•	Ensure all required modules exist.
	•	Select versions (pinned or resolved).
	•	Produce ordered list (topological sort).

Stage 3 — Resolve config + conditions
	•	Create a single config object (pack-level).
	•	For each module/entity/field/index/relationship with when, evaluate the expression.
	•	Remove any disabled entities/fields/relationships/indexes.
	•	Produce the Resolved Manifest:
	•	entities[] (final)
	•	relationships[] (final)
	•	events_catalog[]
	•	attach_points[]
	•	ai_capabilities[]

Stage 4 — Normalize & validate
	•	Validate:
	•	table name uniqueness
	•	field name uniqueness per table
	•	FK targets exist
	•	PK exists
	•	circular FK constraints manageable
	•	Apply standard conventions:
	•	created_at / updated_at triggers (optional)
	•	default UUID generation (dialect-specific)
	•	Optional: generate “soft delete” indexes if deleted_at exists.

Stage 5 — SQL generation
For each entity:
	•	CREATE TABLE
	•	constraints (PK/unique/check)
	•	indexes
For relationships:
	•	ALTER TABLE ADD CONSTRAINT FK (or inline)
For optional features (pgvector):
	•	EXTENSION + vector type support

Output:
	•	schema.sql (full create)
	•	or migration.sql (diff-based if you implement state)

Stage 6 — Runtime wiring artifacts
Generate a “runtime plan” for your services:
	•	Subscriptions for automation.rules that listen to core.events
	•	Workflow triggers from automation.workflow.workflow_triggers
	•	Agent tool registry from ai.agents.ai_tools
	•	Integration endpoints from automation.integrations.inbound_webhooks

This stage produces:
	•	runtime_manifest.json

Stage 7 — Persist build
Write to pack_builds:
	•	resolved_manifest
	•	sql_migration
	•	(optionally) runtime_manifest

3B) Minimal MCP server endpoints (what I’d implement)

Blueprint / Packs
	•	GET /modules
	•	GET /modules/{module_key}/versions
	•	POST /packs (create)
	•	POST /packs/{pack_key}/build → returns {build_id, sql, resolved_manifest, runtime_manifest}

Introspection for no-code UI
	•	GET /packs/{pack_key}/manifest/latest
	•	GET /packs/{pack_key}/schema/latest (SQL)
	•	GET /packs/{pack_key}/catalog/entities
	•	GET /packs/{pack_key}/catalog/events
	•	GET /packs/{pack_key}/catalog/attach-points
	•	GET /packs/{pack_key}/catalog/ai-capabilities

Runtime hooks
	•	POST /events (emit)
	•	POST /webhooks/{path_key} (inbound webhook → event emit)
	•	Worker endpoints (internal):
	•	/workers/rules/run
	•	/workers/scheduler/tick
	•	/workers/agents/run

3C) How the workers run (simple and robust)

Rules worker loop
	1.	Poll events for new entries OR event_deliveries pending.
	2.	Find matching automation_rules by event_name + enabled + tenant.
	3.	Create automation_rule_executions.
	4.	Evaluate conditions.
	5.	Execute actions in order:
	•	webhook.call (use integration_http_action_templates)
	•	ai.run (insert into ai_runs; optionally call model immediately)
	•	event.emit (insert into events)
	•	workflow.start (insert workflow_instance)
	6.	Update status, dead-letter if needed.

Scheduler worker loop
	1.	Poll automation_schedules where next_run_at <= now() AND enabled.
	2.	Create automation_schedule_runs.
	3.	Emit events row (scheduler.tick or the schedule’s event_name).
	4.	Compute next_run_at.

Agents worker loop
	1.	Poll ai_agent_tasks where status=queued.
	2.	Create an ai_runs row for “agent reasoning” (optional).
	3.	Execute tool calls → log to ai_agent_task_steps.
	4.	Write outputs and mark success/failure.

CREATE TABLE ui_projects (
  id uuid PRIMARY KEY,
  name text NOT NULL,
  description text,
  created_at timestamptz NOT NULL DEFAULT now()
);

CREATE TABLE ui_canvases (
  id uuid PRIMARY KEY,
  project_id uuid NOT NULL REFERENCES ui_projects(id) ON DELETE CASCADE,
  name text NOT NULL,
  created_at timestamptz NOT NULL DEFAULT now()
);

-- Each canvas references a pack (optional) and stores layout positions
CREATE TABLE ui_canvas_nodes (
  id uuid PRIMARY KEY,
  canvas_id uuid NOT NULL REFERENCES ui_canvases(id) ON DELETE CASCADE,
  node_type text NOT NULL,          -- "module" | "entity" | "event" | "ai_capability" | "attach_point"
  ref_key text NOT NULL,            -- e.g. module_key or "module:entity" or event_name
  x numeric(10,2) NOT NULL,
  y numeric(10,2) NOT NULL,
  width numeric(10,2),
  height numeric(10,2),
  ui_state jsonb
);

CREATE TABLE ui_canvas_edges (
  id uuid PRIMARY KEY,
  canvas_id uuid NOT NULL REFERENCES ui_canvases(id) ON DELETE CASCADE,
  edge_type text NOT NULL,          -- "dependency" | "relationship" | "emits" | "consumes" | "attaches"
  from_ref text NOT NULL,
  to_ref text NOT NULL,
  ui_state jsonb
);

-- A draft blueprint document being edited (before publishing a version)
CREATE TABLE ui_blueprint_drafts (
  id uuid PRIMARY KEY,
  module_key text NOT NULL,
  base_version text,
  draft_json jsonb NOT NULL,
  status text NOT NULL DEFAULT 'draft',  -- draft|validated|published
  created_at timestamptz NOT NULL DEFAULT now(),
  updated_at timestamptz NOT NULL DEFAULT now()
);

-- Validation results for the UI to show warnings/errors
CREATE TABLE ui_validation_results (
  id uuid PRIMARY KEY,
  draft_id uuid NOT NULL REFERENCES ui_blueprint_drafts(id) ON DELETE CASCADE,
  severity text NOT NULL,           -- info|warning|error
  code text NOT NULL,               -- e.g. "FK_TARGET_MISSING"
  message text NOT NULL,
  pointer text,                     -- json pointer like "/entities/2/fields/4"
  created_at timestamptz NOT NULL DEFAULT now()
);
